056/**
057 * This is the abstract class from which all entropy decoders must
058 * inherit. This class implements the 'MultiResImgData', therefore it has the
059 * concept of a current tile and all operations are performed on the current
060 * tile.
061 *
062 * <P>Default implementations of the methods in 'MultiResImgData' are provided
063 * through the 'MultiResImgDataAdapter' abstract class.
064 *
065 * <P>Sign magnitude representation is used (instead of two's complement) for
066 * the output data. The most significant bit is used for the sign (0 if
067 * positive, 1 if negative). Then the magnitude of the quantized coefficient
068 * is stored in the next most significat bits. The most significant magnitude
069 * bit corresponds to the most significant bit-plane and so on.
070 *
071 * @see MultiResImgData
072 * @see MultiResImgDataAdapter
073 * */
074public abstract class EntropyDecoder extends MultiResImgDataAdapter
075    implements CBlkQuantDataSrcDec {
076
077    /** The prefix for entropy decoder optiojns: 'C' */
078    public final static char OPT_PREFIX = 'C';
079
080    /** The list of parameters that is accepted by the entropy
081     * decoders. They start with 'C'. */
082    private final static String [][] pinfo = {
083        {"Cverber", "[true|false]",
084         "Specifies if the entropy decoder should be verbose about detected "+
085         "errors. If 'true' a message is printed whenever an error is detected.",
086         "true"},
087        {"Cer", "[true|false]",
088         "Specifies if error detection should be performed by the entropy "+
089         "decoder engine. If errors are detected they will be concealed and "+
090         "the resulting distortion will be less important. Note that errors "+
091         "can only be detected if the encoder that generated the data "+
092         "included error resilience information.", "true"},
093    };
094
095    /** The bit stream transport from where to get the compressed data
096     * (the source) */
097    protected CodedCBlkDataSrcDec src;
098
099    /**
100     * Initializes the source of compressed data.
101     *
102     * @param src From where to obtain the compressed data.
103     * */
104    public EntropyDecoder(CodedCBlkDataSrcDec src) {
105        super(src);
106        this.src = src;
107    }
108
109    /**
110     * Returns the subband tree, for the specified tile-component. This method
111     * returns the root element of the subband tree structure, see Subband and
112     * SubbandSyn. The tree comprises all the available resolution levels.
113     *
114     * <P>The number of magnitude bits ('magBits' member variable) for
115     * each subband is not initialized.
116     *
117     * @param t The index of the tile, from 0 to T-1.
118     *
119     * @param c The index of the component, from 0 to C-1.
120     *
121     * @return The root of the tree structure.
122     * */
123
124    public SubbandSyn getSynSubbandTree(int t,int c) {
125        return src.getSynSubbandTree(t,c);
126    }
127
128    /**
129     * Returns the horizontal code-block partition origin. Allowable values
130     * are 0 and 1, nothing else.
131     * */
132    public int getCbULX() {
133        return src.getCbULX();
134    }
135
136    /**
137     * Returns the vertical code-block partition origin. Allowable values are
138     * 0 and 1, nothing else.
139     * */
140    public int getCbULY() {
141        return src.getCbULY();
142    }
143
144
145    /**
146     * Returns the parameters that are used in this class and
147     * implementing classes. It returns a 2D String array. Each of the
148     * 1D arrays is for a different option, and they have 3
149     * elements. The first element is the option name, the second one
150     * is the synopsis and the third one is a long description of what
151     * the parameter is. The synopsis or description may be 'null', in
152     * which case it is assumed that there is no synopsis or
153     * description of the option, respectively. Null may be returned
154     * if no options are supported.
155     *
156     * @return the options name, their synopsis and their explanation,
157     * or null if no options are supported.
158     * */
159    public static String[][] getParameterInfo() {
160        return pinfo;
161    }
162
