047import java.awt.Point;
048
049import jj2000.j2k.wavelet.synthesis.*;
050import jj2000.j2k.image.*;

052/**
053 * This interface defines a source of entropy coded data and methods to
054 * transfer it in a code-block by code-block basis. In each call to
055 * 'geCodeBlock()' a specified coded code-block is returned.
056 *
057 * <P>This interface is the source of data for the entropy decoder. See the
058 * 'EntropyDecoder' class.
059 *
060 * <P>For each coded-code-block the entropy-coded data is returned along with
061 * its truncation point information in a 'DecLyrdCBlk' object.
062 *
063 * @see EntropyDecoder
064 *
065 * @see DecLyrdCBlk
066 *
067 * @see jj2000.j2k.codestream.reader.BitstreamReaderAgent
068 * */
069public interface CodedCBlkDataSrcDec extends InvWTData {
070
071    /**
072     * Returns the specified coded code-block, for the specified component, in
073     * the current tile. The first layer to return is indicated by 'fl'. The
074     * number of layers that is returned depends on 'nl' and the amount of
075     * data available.
076     *
077     * <P>The argument 'fl' is to be used by subsequent calls to this method
078     * for the same code-block. In this way supplamental data can be retrieved
079     * at a later time. The fact that data from more than one layer can be
080     * returned means that several packets from the same code-block, of the
081     * same component, and the same tile, have been concatenated.
082     *
083     * <P>The returned compressed code-block can have its progressive
084     * attribute set. If this attribute is set it means that more data can be
085     * obtained by subsequent calls to this method (subject to transmission
086     * delays, etc). If the progressive attribute is not set it means that the
087     * returned data is all the data that can be obtained for the specified
088     * subblock.
089     *
090     * <P>The compressed code-block is uniquely specified by the current tile,
091     * the component (identified by 'c'), the subband (indentified by 'sb')
092     * and the code-bock vertical and horizontal indexes 'm' and 'n'.
093     *
094     * <P>The 'ulx' and 'uly' members of the returned 'DecLyrdCBlk' object
095     * contain the coordinates of the top-left corner of the block, with
096     * respect to the tile, not the subband.
097     *
098     * @param c The index of the component, from 0 to N-1.
099     *
100     * @param m The vertical index of the code-block to return, in the
101     * specified subband.
102     *
103     * @param n The horizontal index of the code-block to return, in the
104     * specified subband.
105     *
106     * @param sb The subband in whic the requested code-block is.
107     *
108     * @param fl The first layer to return.
109     *
110     * @param nl The number of layers to return, if negative all available
111     * layers are returned, starting at 'fl'.
112     *
113     * @param ccb If not null this object is used to return the compressed
114     * code-block. If null a new object is created and returned. If the data
115     * array in ccb is not null then it can be reused to return the compressed
116     * data.
117     *
118     * @return The compressed code-block, with a certain number of layers
119     * determined by the available data and 'nl'.
120     * */
121    public DecLyrdCBlk getCodeBlock(int c, int m, int n,
122                                         SubbandSyn sb, int fl, int nl,
123                                         DecLyrdCBlk ccb);
124}
