051package jj2000.j2k.entropy.encoder;
052import java.awt.Point;
053
054import jj2000.j2k.wavelet.analysis.*;
055import jj2000.j2k.image.*;
056
057/**
058 * This interface defines a source of entropy coded data and methods to
059 * transfer it in a code-block by code-block basis. In each call to
060 * 'getNextCodeBlock()' a new coded code-block is returned. The code-block are
061 * retruned in no specific-order.
062 *
063 * <P>This interface is the source of data for the rate allocator. See the
064 * 'PostCompRateAllocator' class.
065 *
066 * <P>For each coded-code-block the entropy-coded data is returned along with
067 * the rate-distortion statistics in a 'CBlkRateDistStats' object.
068 *
069 * @see PostCompRateAllocator
070 *
071 * @see CBlkRateDistStats
072 *
073 * @see EntropyCoder
074 * */
075public interface CodedCBlkDataSrcEnc extends ForwWTDataProps {
076
077    /**
078     * Returns the next coded code-block in the current tile for the specified
079     * component, as a copy (see below). The order in which code-blocks are
080     * returned is not specified. However each code-block is returned only
081     * once and all code-blocks will be returned if the method is called 'N'
082     * times, where 'N' is the number of code-blocks in the tile. After all
083     * the code-blocks have been returned for the current tile calls to this
084     * method will return 'null'.
085     *
086     * <P>When changing the current tile (through 'setTile()' or 'nextTile()')
087     * this method will always return the first code-block, as if this method
088     * was never called before for the new current tile.
089     *
090     * <P>The data returned by this method is always a copy of the internal
091     * data of this object, if any, and it can be modified "in place" without
092     * any problems after being returned.
093     *
094     * @param c The component for which to return the next code-block.
095     *
096     * @param ccb If non-null this object might be used in returning the coded
097     * code-block in this or any subsequent call to this method. If null a new
098     * one is created and returned. If the 'data' array of 'cbb' is not null
099     * it may be reused to return the compressed data.
100     *
101     * @return The next coded code-block in the current tile for component
102     * 'n', or null if all code-blocks for the current tile have been
103     * returned.
104     *
105     * @see CBlkRateDistStats
106     *
107     *
108     * */
109    public CBlkRateDistStats getNextCodeBlock(int c, CBlkRateDistStats ccb);
110
111    /**
112     * Returns the width of a packet for the specified tile-
113     * component and resolution level.
114     *
115     * @param t The tile
116     *
117     * @param c The component
118     *
119     * @param r The resolution level
120     *
121     * @return The width of a packet for the specified tile-
122     * component and resolution level.
123     *
124     * */
125    public int getPPX(int t, int c, int r);
126
127    /**
128     * Returns the height of a packet for the specified tile-
129     * component and resolution level.
130     *
131     * @param t The tile
132     *
133     * @param c The component
134     *
135     * @param r The resolution level
136     *
137     * @return The height of a packet for the specified tile-
138     * component and resolution level.
139     *
140     * */
141    public int getPPY(int t, int c, int r);
142
143    /**
144     * Returns true if the precinct partition is used for the
145     * specified component and tile, returns false otherwise
146     *
147     * @param c The component
148     *
149     * @param t The tile
150     *
151     */
152    public boolean precinctPartitionUsed(int c, int t);
153}
