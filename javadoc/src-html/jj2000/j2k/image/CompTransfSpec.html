049import jj2000.j2k.image.invcomptransf.*;
050import jj2000.j2k.wavelet.*;
051import jj2000.j2k.util.*;
052import jj2000.j2k.*;
053
054import java.util.*;
055
056/**
057 * This class extends ModuleSpec class in order to hold tile
058 * specifications for component transformation
059 *
060 * @see ModuleSpec
061 *
062 * */
063public class CompTransfSpec extends ModuleSpec{
064
065    /**
066     * Constructs an empty 'CompTransfSpec' with specified number of
067     * tile and components. This constructor is called by the
068     * decoder. Note: The number of component is here for symmetry
069     * purpose. It is useless since only tile specifications are
070     * meaningful.
071     *
072     * @param nt Number of tiles
073     *
074     * @param nc Number of components
075     *
076     * @param type the type of the specification module i.e. tile specific,
077     * component specific or both.
078     *
079     */
080    public CompTransfSpec(int nt, int nc, byte type){
081        super(nt, nc, type);
082    }
083
084
085    /**
086     * Check if component transformation is used in any of the tiles. This
087     * method must not be used by the encoder.
088     *
089     * @return True if a component transformation is used in at least on
090     * tile.
091     *
092     */
093    public boolean isCompTransfUsed(){
094        if( ((Integer)def).intValue() != InvCompTransf.NONE ){
095            return true;
096        }
097
098        if(tileDef!=null){
099            for(int t=nTiles-1; t>=0; t--){
100                if(tileDef[t]!=null &&
101                   ( ((Integer)tileDef[t]).intValue() != InvCompTransf.NONE) ){
102                    return true;
103                }
104            }
105        }
106        return false;
107    }
108}
