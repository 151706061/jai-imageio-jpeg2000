050
051/**
052 * This exception is thrown whenever a feature or functionality that
053 * has not been implemented is calle.
054 *
055 * <P>Its purpose it is to ease the development and testing process. A
056 * class that partially implements its functionality should throw a
057 * <tt>NotImplementedError</tt> when a method that has not yet
058 * been implemented is called.
059 *
060 * <P>This class is made a subclass of <tt>Error</tt> since it should
061 * never be caught by an application. There is no need to declare this
062 * exception in the <tt>throws</tt> clause of a method.
063 *
064 * @see Error
065 */
066public class NotImplementedError extends Error {
067
068    /**
069     * Constructs a new <tt>NotImplementedError</tt> exception with
070     * the default detail message. The message is:
071     *
072     * <P><I>The called method has not been implemented yet. Sorry!</I>
073     *
074     *
075     */
076    public NotImplementedError() {
077        super("The called method has not been implemented yet. Sorry!");
078    }
079
080    /**
081     * Constructs a new <tt>NotImplementedError</tt> exception with
082     * the specified detail message <tt>m</tt>.
083     *
084     * @param m The detail message to use
085     *
086     *
087     */
088    public NotImplementedError(String m) {
089        super(m);
090    }
091}
