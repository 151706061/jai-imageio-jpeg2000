047package jj2000.j2k.codestream;
048
049/**
050 * This interface defines the identifiers for the different bit stream
051 * profiles and progression types.
052 *
053 * <P>Each progressive type has a different number: 'PT_SNR_PROG',
054 * 'PT_RES_PROG', or 'PT_ARB_PROG'. These are the same identifiers are used in
055 * the codestream syntax.
056 *
057 * <P>Each profile identifier is a flag bit. Therefore, several
058 * profiles can appear at the same time.
059 *
060 * <p>This interface defines the constants only. In order to use the constants
061 * in any other class you can either use the fully qualified name (e.g.,
062 * <tt>ProgressionType.LY_RES_COMP_POS_PROG</tt>) or declare this interface in
063 * the implements clause of the class and then access the identifier
064 * directly.</p>
065 * */
066public interface ProgressionType {
067
068    /** The bit stream is Layer/Resolution/Component/Position progressive : 0
069     * */
070    public final static int LY_RES_COMP_POS_PROG = 0;
071
072    /** The bit stream is Resolution/Layer/Component/Position progressive : 1
073     * */
074    public final static int RES_LY_COMP_POS_PROG = 1;
075
076    /** The bit stream is Resolution/Position/Component/Layer progressive : 2
077     * */
078    public final static int RES_POS_COMP_LY_PROG = 2;
079
080    /** The bit stream is Position/Component/Resolution/Layer progressive : 3
081     * */
082    public final static int POS_COMP_RES_LY_PROG = 3;
083
084    /** The bit stream is Component/Position/Resolution/Layer progressive : 4
085     * */
086    public final static int COMP_POS_RES_LY_PROG = 4;
087}
