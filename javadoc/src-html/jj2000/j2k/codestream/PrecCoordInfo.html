044package jj2000.j2k.codestream;
045
046/**
047 * This class is used to store the coordinates of precincts.
048 * */
049public class PrecCoordInfo extends CoordInfo {
050
051    /** Horizontal upper left coordinate in the reference grid */
052    public int xref;
053
054    /** Vertical upper left coordinate on the reference grid */
055    public int yref;
056
057    /**
058     * Constructor. Creates a PrecCoordInfo object.
059     *
060     * @param ulx Horizontal upper left coordinate in the subband
061     *
062     * @param uly Vertical upper left coordinate in the subband
063     *
064     * @param w Precint's width
065     *
066     * @param h Precinct's height
067     *
068     * @param xref The horizontal coordinate on the reference grid
069     *
070     * @param yref The vertical coordinate on the reference grid
071     * */
072    public PrecCoordInfo(int ulx, int uly, int w, int h,
073                        int xref, int yref) {
074        super(ulx, uly, w, h);
075        this.xref = xref;
076        this.yref = yref;
077    }
078
079    /**
080     * Empty Constructor. Creates an empty PrecCoordInfo object.
081     * */
082    public PrecCoordInfo() {
083        super();
084    }
085
086    /**
087     * Returns precinct's information in a String
088     *
089     * @return String with precinct's information
090     * */
091    public String toString() {
092        return super.toString() + ", xref="+xref+", yref="+yref;
093    }
094}
