045package com.sun.media.imageioimpl.plugins.jpeg2000;
046
047import java.awt.Point;
048import java.awt.Rectangle;
049import java.awt.image.Raster;
050import java.awt.image.WritableRaster;
051import java.io.*;
052import javax.imageio.stream.ImageInputStream;
053import jj2000.j2k.codestream.reader.*;
054
055import com.sun.media.imageioimpl.common.SimpleRenderedImage;
056
057public class J2KRenderedImage extends SimpleRenderedImage {
058    private Raster currentTile;
059    private Point currentTileGrid;
060
061    private J2KReadState readState;
062
063    public J2KRenderedImage(ImageInputStream iis,
064                            J2KImageReadParamJava param,
065                            J2KMetadata metadata,
066                            J2KImageReader reader) throws IOException {
067        this(new J2KReadState(iis, param, metadata, reader));
068    }
069
070    public J2KRenderedImage(ImageInputStream iis,
071                            J2KImageReadParamJava param,
072                            J2KImageReader reader) throws IOException {
073        this(new J2KReadState(iis, param, reader));
074    }
075
076    public J2KRenderedImage(J2KReadState readState) {
077        this.readState = readState;
078
079        HeaderDecoder hd = readState.getHeader();
080
081        //determnined by destination
082        Rectangle destinationRegion = readState.getDestinationRegion();
083        width = destinationRegion.width;
084        height = destinationRegion.height;
085        minX = destinationRegion.x;
086        minY = destinationRegion.y;
087
088        Rectangle tile0Rect = readState.getTile0Rect();
089        tileWidth = tile0Rect.width;
090        tileHeight = tile0Rect.height;
091        tileGridXOffset = tile0Rect.x;
092        tileGridYOffset = tile0Rect.y;
093
094        sampleModel = readState.getSampleModel();
095        colorModel = readState.getColorModel();
096    }
097
098    public synchronized Raster getTile(int tileX, int tileY) {
099        if (currentTile != null &&
100            currentTileGrid.x == tileX &&
101            currentTileGrid.y == tileY)
102            return currentTile;
103
104        if (tileX >= getNumXTiles() || tileY >= getNumYTiles())
105            throw new IllegalArgumentException(I18N.getString("J2KReadState1"));
106
107        try {
108            int x = tileXToX(tileX);
109            int y = tileYToY(tileY);
110            currentTile =
111                Raster.createWritableRaster(sampleModel,
112                                            new Point(x, y));
113            currentTile = readState.getTile(tileX,
114                                            tileY,
115                                            (WritableRaster)currentTile);
116        } catch (IOException e) {
117            throw new RuntimeException(e);
118        }
119
120        if (currentTileGrid == null)
121            currentTileGrid = new Point(tileX, tileY);
122        else {
123            currentTileGrid.x = tileX;
124            currentTileGrid.y = tileY;
125        }
126
127        return currentTile;
128    }
129}
