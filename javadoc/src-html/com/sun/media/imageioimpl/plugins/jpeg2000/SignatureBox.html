045package com.sun.media.imageioimpl.plugins.jpeg2000;
046import org.w3c.dom.Node;
047import javax.imageio.metadata.IIOInvalidTreeException;
048import javax.imageio.metadata.IIOMetadataNode;
049
050/** This class is defined to represent a Signature Box of JPEG JP2
051 *  file format.  This type of box has a fixed length of 12, a type of "jP  "
052 *  and a four byte content of 0x0D0A870A, which is used to detects of the
053 *  common file transmission errors which substitutes <CR><LF> with <LF> or
054 *  vice versa.
055 */
056public class SignatureBox extends Box {
057    /** Constructs a <code>SignatureBox</code>. */
058    public SignatureBox() {
059        super(12, 0x6A502020, null);
060    }
061
062    /** Constructs a <code>SignatureBox</code> based on the provided
063     *  <code>org.w3c.dom.Node</code>.
064     */
065    public SignatureBox(Node node) throws IIOInvalidTreeException {
066        super(node);
067    }
068
069    /** Constructs a <code>SignatureBox</code> based on the provided
070     *  byte array.
071     */
072    public SignatureBox(byte[] data) throws IIOInvalidTreeException {
073        super(12, 0x6A502020, data);
074    }
075
076    /** Creates an <code>IIOMetadataNode</code> from this signature
077     *  box.  The format of this node is defined in the XML dtd and xsd
078     *  for the JP2 image file.
079     */
080    public IIOMetadataNode getNativeNode() {
081        IIOMetadataNode node = new IIOMetadataNode(Box.getName(getType()));
082        setDefaultAttributes(node);
083        node.setAttribute("Signature", Integer.toString(0x0D0A870A));
084        return node;
085    }
086
087    protected void compose() {
088        if (data != null)
089            return;
090        data = new byte[]{(byte)0x0D, (byte)0x0A, (byte)0x87, (byte)0x0A};
091    }
092}
