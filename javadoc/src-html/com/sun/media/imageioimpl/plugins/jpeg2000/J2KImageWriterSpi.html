045package com.sun.media.imageioimpl.plugins.jpeg2000;
046
047import java.awt.image.DataBuffer;
048import java.awt.image.SampleModel;
049
050import javax.imageio.spi.ImageWriterSpi;
051import javax.imageio.spi.ServiceRegistry;
052import javax.imageio.spi.IIORegistry;
053import javax.imageio.ImageWriter;
054import javax.imageio.ImageTypeSpecifier;
055import javax.imageio.IIOException;
056
057import java.awt.image.ColorModel;
058import java.awt.image.IndexColorModel;
059import java.util.Locale;
060
061import com.sun.media.imageioimpl.common.PackageUtil;
062
063public class J2KImageWriterSpi extends ImageWriterSpi {
064    private static String [] readerSpiNames =
065        {"com.sun.media.imageioimpl.plugins.jpeg2000.J2KImageReaderSpi"};
066    private static String[] formatNames =
067        {"jpeg 2000", "JPEG 2000", "jpeg2000", "JPEG2000"};
068    private static String[] extensions =
069        {"jp2"}; // Should add jpx or jpm
070    private static String[] mimeTypes = {"image/jp2", "image/jpeg2000"};
071    private boolean registered = false;
072
073    public J2KImageWriterSpi() {
074        super(PackageUtil.getVendor(),
075              PackageUtil.getVersion(),
076              formatNames,
077              extensions,
078              mimeTypes,
079              "com.sun.media.imageioimpl.plugins.jpeg2000.J2KImageWriter",
080              STANDARD_OUTPUT_TYPE,
081              readerSpiNames,
082              false,
083              null, null,
084              null, null,
085              true,
086              "com_sun_media_imageio_plugins_jpeg2000_image_1.0",
087              "com.sun.media.imageioimpl.plugins.jpeg2000.J2KMetadataFormat",
088              null, null);
089    }
090
091    public String getDescription(Locale locale) {
092        String desc = PackageUtil.getSpecificationTitle() + 
093            " JPEG 2000 Image Writer";
094        return desc;
095    }
096
097    public void onRegistration(ServiceRegistry registry,
098                               Class category) {
099        if (registered) {
100            return;
101        }
102        
103        registered = true;
104
105        // Set pairwise ordering to give codecLib writer precedence.
106        Class codecLibWriterSPIClass = null;
107        try {
108            codecLibWriterSPIClass =
109                Class.forName("com.sun.media.imageioimpl.plugins.jpeg2000.J2KImageWriterCodecLibSpi");
110        } catch(Throwable t) {
111            // Ignore it.
112        }
113
114        if(codecLibWriterSPIClass != null) {
115            Object codecLibWriterSPI =
116                registry.getServiceProviderByClass(codecLibWriterSPIClass);
117            if(codecLibWriterSPI != null) {
118                registry.setOrdering(category, codecLibWriterSPI, this);
119            }
120        }
121    }
122
123    public boolean canEncodeImage(ImageTypeSpecifier type) {
124        SampleModel sm = type.getSampleModel();
125        if (sm.getNumBands() > 16384)
126            return false;
127        if (sm.getDataType() < DataBuffer.TYPE_BYTE ||
128            sm.getDataType() > DataBuffer.TYPE_INT)
129            return false;
130        return true;
131    }
132
133    public ImageWriter createWriterInstance(Object extension)
134        throws IIOException {
135        return new J2KImageWriter(this);
136    }
137}
