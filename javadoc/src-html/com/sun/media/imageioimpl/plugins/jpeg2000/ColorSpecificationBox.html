045package com.sun.media.imageioimpl.plugins.jpeg2000;
046
047import java.awt.color.ICC_Profile;
048import javax.imageio.metadata.IIOInvalidTreeException;
049import javax.imageio.metadata.IIOMetadataNode;
050import org.w3c.dom.Node;
051import org.w3c.dom.NodeList;
052
053/** This class is defined to represent a Color Specification Box of JPEG JP2
054 *  file format.  A Channel Definition Box has a length, and a fixed type
055 *  of "colr".  Its content contains the method to define the color space,
056 *  the precedence and approximation accuracy (0 for JP2 files), the
057 *  enumerated color space, and the ICC color profile if any.
058 */
059public class ColorSpecificationBox extends Box {
060    /** The enumerated color space defined in JP2 file format. */
061    public static final int ECS_sRGB = 16;
062    public static final int ECS_GRAY = 17;
063    public static final int ECS_YCC = 18;
064
065    /** Cache the element names for this box's xml definition */
066    private static String[] elementNames = {"Method", "Precedence",
067                                            "ApproximationAccuracy",
068                                            "EnumeratedColorSpace",
069                                            "ICCProfile"};
070
071    /** This method will be called by the getNativeNodeForSimpleBox of the
072     *  class Box to get the element names.
073     */
074    public static String[] getElementNames() {
075        return elementNames;
076    }
077
078    /** The elements' values. */
079    private byte method;
080    private byte precedence;
081    private byte approximation;
082    private int ecs;
083    private ICC_Profile profile;
084
085    /** Computes the length of this box when profile is present. */
086    private static int computeLength(byte m, ICC_Profile profile) {
087        int ret = 15;
088        if (m == 2 && profile != null) {
089            ret += profile.getData().length;
090        }
091        return ret;
092    }
093
094    /** Creates a <code>ColorSpecificationBox</code> from the provided data
095     *  elements.
096     */
097    public ColorSpecificationBox(byte m, byte p, byte a, int ecs,
098                                 ICC_Profile profile) {
099        super(computeLength(m, profile), 0x636F6C72, null);
100        this.method = m;
101        this.precedence = p;
102        this.approximation = a;
103        this.ecs = ecs;
104        this.profile = profile;
105    }
106
107    /** Creates a <code>ColorSpecificationBox</code> from the provided byte
108     *  array.
109     */
110    public ColorSpecificationBox(byte[] data) {
111        super(8 + data.length, 0x636F6C72, data);
112    }
113
114    /** Constructs a <code>ColorSpecificationBox</code> based on the provided
115     *  <code>org.w3c.dom.Node</code>.
116     */
117    public ColorSpecificationBox(Node node) throws IIOInvalidTreeException {
118        super(node);
119        NodeList children = node.getChildNodes();
120
121        for (int i = 0; i < children.getLength(); i++) {
122            Node child = children.item(i);
123            String name = child.getNodeName();
124
125            if ("Method".equals(name)) {
126                method = Box.getByteElementValue(child);
127            }
128
129            if ("Precedence".equals(name)) {
130                precedence = Box.getByteElementValue(child);
131            }
132
133            if ("ApproximationAccuracy".equals(name)) {
134                approximation = Box.getByteElementValue(child);
135            }
136
137            if ("EnumeratedColorSpace".equals(name)) {
138                ecs = Box.getIntElementValue(child);
139            }
140
141            if ("ICCProfile".equals(name)) {
142                if (child instanceof IIOMetadataNode)
143                    profile =
144                        (ICC_Profile)((IIOMetadataNode)child).getUserObject();
145                else {
146                    String value = node.getNodeValue();
147                    if (value != null)
148                        profile = ICC_Profile.getInstance(Box.parseByteArray(value));
149                }
150            }
151        }
152    }
153
154    /** Returns the method to define the color space. */
155    public byte getMethod() {
156        return method;
157    }
158
159    /** Returns <code>Precedence</code>. */
160    public byte getPrecedence() {
161        return precedence;
162    }
163
164    /** Returns <code>ApproximationAccuracy</code>. */
165    public byte getApproximationAccuracy() {
166        return approximation;
167    }
168
169    /** Returns the enumerated color space. */
170    public int getEnumeratedColorSpace() {
171        return ecs;
172    }
173
174    /** Returns the ICC color profile in this color specification box. */
175    public ICC_Profile getICCProfile() {
176        return profile;
177    }
178
179    /** Creates an <code>IIOMetadataNode</code> from this color specification
180     *  box.  The format of this node is defined in the XML dtd and xsd
181     *  for the JP2 image file.
182     */
183    public IIOMetadataNode getNativeNode() {
184        return getNativeNodeForSimpleBox();
185    }
186
187    protected void parse(byte[] data) {
188        method = data[0];
189        precedence = data[1];
190        approximation = data[2];
191        if (method == 2) {
192            byte[] proData = new byte[data.length - 3];
193            System.arraycopy(data, 3, proData, 0, data.length - 3);
194            profile = ICC_Profile.getInstance(proData);
195        } else
196            ecs = ((data[3] & 0xFF) << 24) | ((data[4] & 0xFF) << 16) |
197                  ((data[5] & 0xFF) << 8) | (data[6] & 0xFF);
198
199    }
200
201    protected void compose() {
202        if (data != null)
203            return;
204        int len = 7;
205        byte[] profileData = null;
206        if (profile != null) {
207            profileData = profile.getData();
208            len += profileData.length;
209        }
210
211        data = new byte[len];
212
213        data[0] = (byte)method;
214        data[1] = (byte)precedence;
215        data[2] = (byte)approximation;
216
217        copyInt(data, 3, ecs);
218
219        if (profile != null)
220            System.arraycopy(profileData, 0, data, 7, len - 7);
221    }
222}
