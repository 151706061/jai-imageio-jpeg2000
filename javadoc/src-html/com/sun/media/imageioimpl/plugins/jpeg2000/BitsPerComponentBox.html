046package com.sun.media.imageioimpl.plugins.jpeg2000;
047
048import javax.imageio.metadata.IIOInvalidTreeException;
049import javax.imageio.metadata.IIOMetadataNode;
050import org.w3c.dom.Node;
051import org.w3c.dom.NodeList;
052
053/** This class is defined to represent a Bits Per Component Box of JPEG
054 *  JP2 file format.  A Bits Per Component box has a length, and a fixed
055 *  type of "bpcc".  Its content is a byte array containing the bit
056 *  depths of the color components.
057 *
058 *  This box is necessary only when the bit depth are not identical for all
059 *  the components.
060 */
061public class BitsPerComponentBox extends Box {
062    /** Counstructs a <code>BitsPerComponentBox</code> from the provided
063     *  byte array containing the bit depths of each color component.
064     */
065    public BitsPerComponentBox(byte[] bitDepth) {
066        super(8 + bitDepth.length,  0x62706363, null);
067        data = bitDepth;
068    }
069
070    /** Constructs a <code>BitsPerComponentBox</code> based on the provide
071     *  <code>org.w3c.dom.Node</code>.
072     */
073    public BitsPerComponentBox(Node node) throws IIOInvalidTreeException {
074        super(node);
075        NodeList children = node.getChildNodes();
076       
077        for (int i = 0; i < children.getLength(); i++) {
078            Node child = children.item(i);
079            String name = child.getNodeName();
080           
081            if ("BitDepth".equals(name)) {
082                data = Box.getByteArrayElementValue(child);
083            }
084        }
085    }
086
087    /** Returns the bit depths for all the image components. */
088    public byte[] getBitDepth() {
089        return data;
090    }
091}
