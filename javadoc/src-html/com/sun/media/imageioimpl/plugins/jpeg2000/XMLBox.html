045package com.sun.media.imageioimpl.plugins.jpeg2000;
046
047import javax.imageio.metadata.IIOInvalidTreeException;
048import javax.imageio.metadata.IIOMetadataNode;
049import org.w3c.dom.Node;
050import org.w3c.dom.NodeList;
051
052/** This class is defined to represent a XML box of JPEG JP2
053 *  file format.  This type of box has a length, a type of "xml ".  Its
054 *  content is a text string of a XML instance.
055 */
056public class XMLBox extends Box {
057    /** Cache the element names for this box's xml definition */
058    private static String[] elementNames = {"Content"};
059
060    /** This method will be called by the getNativeNodeForSimpleBox of the
061     *  class Box to get the element names.
062     */
063    public static String[] getElementNames() {
064        return elementNames;
065    }
066
067    /** Create a Box from its content. */
068    public XMLBox(byte[] data) {
069        super(8 + data.length, 0x786D6C20, data);
070    }
071
072    /** Constructs a <code>UUIDListBox</code> based on the provided
073     *  <code>org.w3c.dom.Node</code>.
074     */
075    public XMLBox(Node node) throws IIOInvalidTreeException {
076        super(node);
077        NodeList children = node.getChildNodes();
078
079        for (int i = 0; i < children.getLength(); i++) {
080            Node child = children.item(i);
081            String name = child.getNodeName();
082
083            if ("Content".equals(name)) {
084                String value = child.getNodeValue();
085                if (value != null)
086                    data = value.getBytes();
087                else if (child instanceof IIOMetadataNode) {
088                    value = (String)((IIOMetadataNode)child).getUserObject();
089                    if (value != null)
090                        data = value.getBytes();
091                }
092            }
093        }
094    }
095
096    /** Creates an <code>IIOMetadataNode</code> from this XML
097     *  box.  The format of this node is defined in the XML dtd and xsd
098     *  for the JP2 image file.
099     */
100    public IIOMetadataNode getNativeNode() {
101        try {
102            IIOMetadataNode node = new IIOMetadataNode(Box.getName(getType()));
103            setDefaultAttributes(node);
104            IIOMetadataNode child = new IIOMetadataNode("Content");
105            String value = null;
106            if (data != null)
107                value = new String(data);
108            child.setUserObject(value);
109            child.setNodeValue(value);
110            node.appendChild(child);    
111            return node;
112        } catch (Exception e) {
113            throw new IllegalArgumentException(I18N.getString("Box0"));
114        }
115    }
116}
